128 void read_json_file(std::string path_file_json) {
129
130 // Open the JSON file and parse its contents
131 std::ifstream i(path_file_json);
132 json j;
133 i >> j;
134 _logfile_json << "Read JSON file..." << std::endl;
135
136 // Retrieve file name of the master (true) image
137 if (j["file_name_true_image"] != nullptr) {
138 _file_name_true_image = j.at("file_name_true_image").get<std::string>();
139 _logfile_json << "Set 'file_name' of true image: " << _file_name_true_image << std::endl;
140 }
141 else {
142 _logfile_json << "No value for 'file_name' in json." << std::endl;
143 }
144
145 // Retrieve file name of the image points to be referenced
146 /*if (j["file_name_image_points"] != nullptr) {
147 _file_name_image_points = j.at("file_name_image_points").get<std::string>();
148 _logfile_json << "Set 'file_name' of image points file: " << _file_name_image_points << std::endl;
149 }
150 else {
151 _logfile_json << "No value for 'file_name_image_points' in json." << std::endl;
152 }*/
153
154 // Retrieve file name of the image points to be referenced
155 if (j.contains("file_name_image_points") && j["file_name_image_points"].is_array()) {
156 _file_name_image_points = j["file_name_image_points"].get<std::vector<std::string>>();
157 }
158 else {
159 _logfile_json << "No value for 'file_name_image_points' in json." << std::endl;
160 }
161
162 // Read and set the focal length (in mm) if available
163 if (j["focal_length_mm"] != nullptr) {
164 _principal_distance = std::stod(j.at("focal_length_mm").get<std::string>());
165 _logfile_json << "Set 'focal_length_mm' ck: " << _principal_distance << " [mm]" << std::endl;
166 }
167 else {
168 _logfile_json << "No value for 'focal_length_mm' in json." << std::endl;
169 }
170
171 // Retrieve the horizontal and vertical view angles
172 if (j["view_angle_x"] != nullptr && j["view_angle_y"] != nullptr) {
173 _view_angle_half_H = std::stod(j.at("view_angle_x").get<std::string>());
174 _view_angle_half_V = std::stod(j.at("view_angle_y").get<std::string>());
175 _logfile_json << "Set (full) opening angles 'view_angle_x'/'view_angle_y' (H/V): "
176 << _view_angle_half_H << "/" << _view_angle_half_V << " [°]" << std::endl;
177 }
178 else {
179 _logfile_json << "No information about view angle horizontal and/or vertical of camera available" << std::endl;
180 }
181
182 // Set the initial projection center coordinates
183 if (j["X0_x"] != nullptr && j["X0_y"] != nullptr && j["X0_z"] != nullptr) {
184 _X0_x = std::stod(j.at("X0_x").get<std::string>());
185 _X0_y = std::stod(j.at("X0_y").get<std::string>());
186 _X0_z = std::stod(j.at("X0_z").get<std::string>());
187 _logfile_json << "Set initial projection center 'X0 (x,y,z)': " << _X0_x << " [m], " << _X0_y << " [m], " << _X0_z << " [m]" << std::endl;
188 }
189 else {
190 _logfile_json << "No value for 'X0 (x,y,z)' in json." << std::endl;
191 }
192
193 // Set the orientation angles for azimuth, pitch, and roll if available
194 if (j["azimuth"] != nullptr && j["pitch"] != nullptr && j["roll"] != nullptr) {
195 _azimuth = std::stod(j.at("azimuth").get<std::string>());
196 _pitch = std::stod(j.at("pitch").get<std::string>());
197 _roll = std::stod(j.at("roll").get<std::string>());
198 _logfile_json << "Set initial rotation angles (Euler, in deg): 'azimuth': " << _azimuth
199 << ", 'pitch': " << _pitch << ", 'roll': " << _roll << std::endl;
200 }
201 else {
202 _logfile_json << "No value for 'azimuth', 'pitch', 'roll' in json." << std::endl;
203 }
204
205 // Set the pixel size in millimeters
206 if (j["pixel_size_mm"] != nullptr) {
207 _pix_size = std::stod(j.at("pixel_size_mm").get<std::string>());
208 _logfile_json << "Set 'pixel_size_mm': " << _pix_size << " [mm]" << std::endl;
209 }
210 else {
211 _logfile_json << "No value for 'pixel_size_mm' in json, Set default: " << _pix_size << " [mm]" << std::endl;
212 }
213
214 // Set the location accuracy
215 if (j["loc_accuracy"] != nullptr) {
216 _loc_acc_X0_z = std::stod(j.at("loc_accuracy").get<std::string>());
217 _loc_acc_X0_xy = _loc_acc_X0_z;
218 _frustum->set_bb_offset_X0_xy(_loc_acc_X0_xy);
219 _frustum->set_bb_offset_X0_z(_loc_acc_X0_z);
220 _logfile_json << "Set 'loc_accuracy'. Noise position location (XY): " << _loc_acc_X0_xy
221 << " [m]" << std::endl;
222 _logfile_json << "Set 'loc_accuracy'. Noise position height (Z): " << _loc_acc_X0_z << " [m]" << std::endl;
223 }
224 else {
225 _logfile_json << "No value for 'loc_accuracy' in json." << std::endl;
226 }
227
228 // Define the maximum distance to the projection center
229 if (j["max_dist_to_X0"] != nullptr) {
230 _max_dist_to_X0 = std::stod(j.at("max_dist_to_X0").get<std::string>());
231 _logfile_json << "Set 'max_dist_to_X0': " << _max_dist_to_X0 << " [m]" << std::endl;
232 }
233 else {
234 _logfile_json << "No value for 'max_dist_to_X0' in json, set default: " << _max_dist_to_X0 << " [m]" << std::endl;
235 }
236
237 // Define the minimum distance from the projection center
238 if (j["min_dist_to_X0"] != nullptr) {
239 _min_dist_to_X0 = std::stod(j.at("min_dist_to_X0").get<std::string>());
240 _logfile_json << "Set 'min_dist_to_X0': " << _min_dist_to_X0 << " [m]" << std::endl;
241 }
242 else {
243 _logfile_json << "No value for 'min_dist_to_X0' in json, set default: " << _min_dist_to_X0 << " [m]" << std::endl;
244 }
245
246 // Load the true image from the specified file path
247 fs::path path_file_trueImage = fs::path(path_file_json).parent_path() / _file_name_true_image;
248 true_image = cv::imread(path_file_trueImage.string());
249 _logfile_json << "Loaded true image from path: " << path_file_trueImage << ". Image size: " << true_image.size() << std::endl;
250
251 // Lade Bildpunkte zur Skalierung aus mehreren Dateien
252 if (!_file_name_image_points.empty()) {
253 for (const auto& file_name : _file_name_image_points) {
254 fs::path path_file_image_points = fs::path(path_file_json).parent_path() / file_name;
255 std::ifstream inputStream(path_file_image_points);
256
257 if (!inputStream.is_open()) {
258 _logfile_json << "Failed to open file: " << path_file_image_points << std::endl;
259 continue;
260 }
261
262 // Vektor zur Speicherung der Punkte aus der aktuellen Datei
263 std::vector<cv::Point2d> points_from_file;
264 double x, y, z;
265 char sep;
266
267 // Lese Punkte aus der Datei und füge sie in den Vektor ein
268 while (inputStream >> x >> sep >> y >> sep >> z) {
269 points_from_file.emplace_back(x, y);
270 }
271
272 // Füge die Punkte dieser Datei dem Hauptvektor hinzu
273 _pts_image_points_2D->push_back(points_from_file);
274
275 // Ausgabe in das Logfile
276 _logfile_json << "Loaded " << points_from_file.size()
277 << " image points from: " << path_file_image_points << std::endl;
278 }
279
280 // Gesamtanzahl der geladenen Bildpunkte über alle Dateien
281 _logfile_json << "Total image points count: " << _pts_image_points_2D->size() << std::endl;
282 }
283 else {
284 _logfile_json << "No image points to be scaled given." << std::endl;
285 }
286
287 // Adjust the view angles for frustum calculations
288 _view_angle_half_H /= 2.0;
289 _view_angle_half_V /= 2.0;
290 _frustum->set_view_angles(_view_angle_half_H, _view_angle_half_V);
291 _logfile_json << "Updated (half) view angles for frustum calculation (in deg): H: " << _view_angle_half_H << ", V: " << _view_angle_half_V << std::endl;
292
293 // Apply coordinate shifts for efficiency
294 _shift_x = _X0_x;
295 _shift_y = _X0_y;
296 _shift_z = _X0_z;
297 _X0_x -= _shift_x;
298 _X0_y -= _shift_y;
299 _X0_z -= _shift_z;
300 _logfile_json << "Applied shift to coordinates X0 for efficiency. shift_x: " << std::fixed << _shift_x << " [m], shift_y: " << _shift_y << " [m], shift_z: " << _shift_z << " [m]" << std::endl; //Use std::fixed floating-point notation for formatting
301
302 _frustum->set_X0_Cam_World(_X0_x, _X0_y, _X0_z); //update frustum X0
303 _frustum->calculate_rotation_matrix_rzxy(_azimuth, _roll, _pitch); //update frustum rotM
304
305 // Precompute trigonometric values for rotation matrix based on azimuth, roll, and pitch angles
306 double rad_azimuth = _azimuth * M_PI / 180.0;
307 double rad_roll = _roll * M_PI / 180.0;
308 double rad_pitch = _pitch * M_PI / 180.0;
309
310 double Cz = cos(rad_azimuth);
311 double Sz = sin(rad_azimuth);
312 double Cy = cos(rad_roll);
313 double Sy = sin(rad_roll);
314 double Cx = cos(rad_pitch);
315 double Sx = sin(rad_pitch);
316
317 // Construct the rotation matrix _rotM (Rxyz) with rotations in the order of Rz (azimuth), Ry (roll), then Rx (pitch).
318 // Rotational axes: x-axis (pitch), y-axis (roll), z-axis (azimuth).
319 _rotM[0] = Cy * Cz; _rotM[3] = -Cy * Sz; _rotM[6] = Sy;
320 _rotM[1] = Sx * Sy * Cz + Cx * Sz; _rotM[4] = -Sx * Sy * Sz + Cx * Cz; _rotM[7] = -Sx * Cy;
321 _rotM[2] = -Cx * Sy * Cz + Sx * Sz; _rotM[5] = Cx * Sy * Sz + Sx * Cz; _rotM[8] = Cx * Cy;
322
323 // Log the constructed rotation matrix for debugging
324 _logfile->append(TAG + "RotM:");
325 _logfile->append("\t\t" + std::to_string(_rotM[0]) + " " + std::to_string(_rotM[3]) + " " + std::to_string(_rotM[6]), 4);
326 _logfile->append("\t\t" + std::to_string(_rotM[1]) + " " + std::to_string(_rotM[4]) + " " + std::to_string(_rotM[7]), 4);
327 _logfile->append("\t\t" + std::to_string(_rotM[2]) + " " + std::to_string(_rotM[5]) + " " + std::to_string(_rotM[8]), 4);
328 _logfile->append("");
329
330 // Set maximum depth of the point cloud for projection within the frustum
331 _frustum->set_frustum_depth(_max_dist_to_X0);
332
333 // Calculate the view frustum based on updated parameters
334 _frustum->calculate_view_frustum();
335
336 // Push log content to logfile
337 std::string line, lineErr;
338 _logfile->append(""); // Add an empty line before JSON content, if needed for separation
339 while (std::getline(_logfile_json, line)) { // Read and log each line from _logfile_json with a "JSON:" prefix
340 _logfile->append("JSON:\t\t" + line);
341 }
342 }

358 bool run_lightglue_image_matching(const cv::Mat& in_true_image, const cv::Mat& in_synth_image) {
359 // Check if both input images are valid
360 if (in_true_image.empty() || in_synth_image.empty()) {
361 _logfile->append("Cannot run LightGlue image matching due to missing images.");
362 return false;
363 }
364 else {
365 _logfile->append("Running LightGlue image matching.");
366 }
367
368 // Define the directory path for input data and result storage
369 fs::path path_directory_myData = get_path_working_directory() / "myData";
370 fs::create_directory(path_directory_myData); // Generate 'myData' directory if it doesn't exist
371
372 // Define file paths for the image list and individual image files
373 fs::path path_file_imagelist_to_match = path_directory_myData / "image_file_list.txt";
374 fs::path path_file_trueImage_to_match = path_directory_myData / "true_image.jpg";
375 fs::path path_file_synthImage_to_match = path_directory_myData / "synth_image.jpg";
376 _path_file_output_lightglue_matches = path_directory_myData / "kpts.txt";
377
378 // Save the images to the 'myData' directory
379 cv::imwrite(path_file_trueImage_to_match.string(), in_true_image);
380 cv::imwrite(path_file_synthImage_to_match.string(), in_synth_image);
381
382 // Wait until the images are fully saved by checking their file sizes
383 while (
384 Utils::calculate_file_size(path_file_trueImage_to_match) == NULL ||
385 Utils::calculate_file_size(path_file_synthImage_to_match) == NULL ||
386 Utils::calculate_file_size(path_file_trueImage_to_match.string()) < 1 ||
387 Utils::calculate_file_size(path_file_synthImage_to_match) < 1) {
388 // Wait until both images have been completely written to disk
389 }
390
391 // Open an output file stream to write the image list file
392 std::ofstream myFileImageList(path_file_imagelist_to_match);
393 myFileImageList << path_file_trueImage_to_match << std::endl;
394 myFileImageList << path_file_synthImage_to_match << std::endl;
395 myFileImageList.close();
396
397 // Ensure the image list file is saved before proceeding
398 while (Utils::calculate_file_size(path_file_imagelist_to_match) == NULL) {
399 // Wait until the image list file has been completely written to disk
400 }
401
402 // Prepare to execute the LightGlue Python script with arguments
403 std::string python_executable = "python"; // Adjust this to "python3" or the path to your Python interpreter as needed
404
405 // Construct the command to execute the Python script with appropriate arguments
406 std::string command = python_executable + " " + this->get_path_python_script_lightglue().string() +
407 " --left_image " + path_file_trueImage_to_match.string() +
408 " --right_image " + path_file_synthImage_to_match.string() +
409 " --output_dir " + path_directory_myData.string();
410
411 // Execute the command to run the Python script
412 int result = std::system(command.c_str());
413
414 // Check if the Python script executed successfully
415 if (result != 0) {
416 std::cerr << "Error executing Python script." << std::endl;
417 return false;
418 }
419
420 _logfile->append("Python-based LightGlue image matching executed successfully.");
421 return true;
422 }
